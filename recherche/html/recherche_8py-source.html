00185 ## Algorithme de recherche
00186 #
00187 # Cette fonction de recherche permet de faire trois types de recherche:
00188 #
00189 # \li \c DFS pour la recherche en profondeur d'abord.
00190 # \li \c BFS pour la recherche en largeur d'abord.
00191 # \li \c A* pour la recherche heuristique
00192 #
00193 # L'algorithme de recherche est le suivant :
00194 #
00195 # @code
00196 # Recherche(Noeud_depart, Noeud_but, methode)
00197 #       1. Q <- [Noeud_depart] 
00198 #       2. while Q non vide:
00199 #       3.       n <- premier(Q), Q <- reste(Q)
00200 #       4.       if n est un noeud but:
00201 #       5.         return n
00202 #   6.   else:
00203 #   7.     S <- successeurs de n
00204 #   8.     Q <- ajouterSuccesseurs(Q,S,methode)
00205 #       9. return ECHEC.
00206 # @endcode
00207 #
00208 # @param[in] depart ville de depart
00209 # @param[in] but ville d'arrivee
00210 # @param[in] methode la methode utilisee (DFS,BFS,A*)
00211 # @return n si c'est un noeud but, ECHEC sinon
00212 # @note Le seul changement entre les diverses methodes de recherche est la
00213 #          facon d'ajouter les successeurs dans Q.
00214 #          \li \c DFS on ajoute S+Q
00215 #               \li \c BFS on ajoute Q+S
00216 #               \li \c A* on ajoute Q+S et on trie par ordre croissant de \f$f_{n}\f$
00217 def recherche( depart, but, methode ):
00218         
00219         # initialise la liste avec le noeud de depart
00220         Q=[libRecherche.Noeud(depart)]
00221         
00222         # initialise le compteur d'iterations
00223         nbIteration=1
00224         
00225         # boucle jusqu'a ce que Q soit vide
00226         while len(Q) > 0:
00227                 
00228                 # selectionne le premier noeud de la liste 
00229                 n = Q.pop(0)
00230                 
00231                 # si on a une solution, on l'imprime et la retourne
00232                 if n.estUneSolution(but):
00233                         print "Solution a l'iteration ",nbIteration,":",n
00234                         return n
00235                 
00236                 # sinon, on augmente le compteur d'iterations et on ajoute
00237                 # les successeurs dans la liste Q selon la methode de recherche
00238                 else:
00239                         nbIteration+=1
00240                         print "Iteration ", nbIteration, ":", n
00241                         S = n.successeurs(but)
00242                         Q = ajouteSuccesseurs(Q, S, methode)
00243                                 
00244         # plus de noeud dans la liste Q
00245         return ECHEC

00248 ## Algorithme de recherche optimisee
00249 #
00250 # L'algorithme de recherche optimisee est tres similaire a celui d'une recherche
00251 # normale, la grande difference etant que l'on ne visite un noeud qu'une seule 
00252 # fois. Pour cela, on garde la liste C qui contient tous les noeuds deja
00253 # visites. 
00254 #
00255 # @code
00256 #RechercheOptimisee(Noeud_depart, Noeud_but, methode)
00257 #       1.  Q <- [Noeud_depart] 
00258 #       2.  C <- [] 
00259 #       3.  while Q non vide:
00260 #       4.        n <- premier(Q), Q <- reste(Q)
00261 #       5.        if n est un noeud but:
00262 #   6.      return n
00263 #   7.    else:
00264 #   8.      if (n not in C and methode is not "A*") or 
00265 #   ...(n=n' in C and f(n)<f(n') and methode is  "A*"):
00266 #   9.            S <- successeurs de n
00267 #   10.           Q <- ajouterSuccesseurs(Q,S,methode)
00268 #   11.           ajoute n dans C
00269 #       12. return ECHEC
00270 # @endcode
00271 #
00272 # @param[in] depart ville de depart
00273 # @param[in] but ville d'arrivee
00274 # @param[in] methode la methode utilisee (DFS,BFS,A*)
00275 # @note Une variante est necessaire pour A*. En effet, un noeud peut etre 
00276 #          deja contenu dans la liste. Cependant, si son cout est superieur a
00277 #          celui trouve par un autre chemin, on met a jour celui-ci
00278 def rechercheOptimisee(depart, but, methode):
00279         
00280         # liste initiale contenant les noeuds en cours. initialisee
00281         # avec le noeud de depart
00282         Q =[libRecherche.Noeud(depart)]
00283         
00284         # liste des noeuds deja visites 
00285         C = libRecherche.Noeuds()
00286         
00287         # initialise le compteur d'iterations
00288         nbIteration=1
00289         
00290         # itere jusqu'a ce que Q soit vide ou 
00291         # que l'on trouve un noeud but
00292         while len(Q) > 0:
00293                 
00294                 # extrait le premier noeud de la liste 
00295                 n = Q.pop(0)
00296                 
00297                 # si celui-ci est une solution, on l'imprime et le retourne 
00298                 if n.estUneSolution(but):
00299                         print "Solution a l'iteration ",nbIteration,":",n
00300                         return n
00301                 
00302                 # sinon, on parcourt ses successeurs en s'assurant que le noeud
00303                 # n n'existe pas deja dans la liste ou, si c'est le cas, a un cout 
00304                 # inferieur
00305                 else:
00306                         if (not C.contient(n) and methode is not "A*") or (C.contient(n) 
00307                                 and C.coutFStrictementInferieur(n) and methode is "A*")  :
00308                                 
00309                                 # affiche l'iteration en cours
00310                                 print "Iteration ", nbIteration, ":", n
00311                                 
00312                                 # incremente le compteur d'iterations
00313                                 nbIteration+=1
00314                                 
00315                                 # ajoute le noeud dans la liste des noeuds visites 
00316                                 C.ajouteNoeud( n )
00317                                 
00318                                 # ajoute ses successeurs selon la methode choisie 
00319                                 S=n.successeurs( but )
00320                                 Q=ajouteSuccesseurs( Q, S, methode )
00321                                         
00322         # plus d'elements dans Q
00323         return ECHEC

00326 ## Ajoute les successeurs S dans la liste Q
00327 #
00328 # Ajoute les succeseurs S a Q suivant la methode de recherche.
00329 #
00330 # @param[in] Q la liste initiale
00331 # @param[in] S la liste des successeurs
00332 # @param[in] methode la methode de recherche { DFS, BFS, A* }
00333 def ajouteSuccesseurs(Q,S,methode):
00334         
00335         # DFS - ajoute les successeurs en tete de liste
00336         if methode=="DFS":
00337                 return S+Q
00338         
00339         # BFS - ajoute les successeurs en queue de liste 
00340         elif methode=="BFS":
00341                 return Q+S
00342         
00343         # A* - ajoute les successeurs en queue de liste et trie la
00344         #         liste par ordre decroissant de cout 
00345         elif methode=="A*":
00346                 Q = Q+S
00347                 Q.sort(key = lambda n : n.coutF)
00348                 return Q
00349         
00350         # La methode n'existe pas 
00351         else:
00352                 print "Methode ", methode," inconnue"
