00129 ## Effectue un chainage avant simple a partir des faits initiaux et de regles donnees
00130 #
00131 # La fonction de chainage avant simple deduit tous les faits possibles de
00132 # deduire a partir des faits initiaux et des regles donnes. Une fois que les 
00133 # consequences deduites sont inscrites dans la bases de donnees, il faudra
00134 # de nouveau verifier si ces nouveaux faits permettent de declencher de
00135 # nouvelles regles. Le processus se poursuit jusqu'a ce qu'il n'y ait plus
00136 # de fait a deduire. Le pseudo-code de l'algorithme de chainage avant est le
00137 # suivant:
00138 #
00139 # @code
00140 #    1. Procedure Chainage-Simple(F,R)
00141 #    2. Q <- faits de depart F
00142 #    3. while Q pas vide {
00143 #    4.  q <- premier(Q), Q <- reste(Q)
00144 #    5.  if q pas dans la base de donnees then 
00145 #    6.                 ajouter q a la base de donnees et imprimer q
00146 #    7.  for chaque regle r de la base de regles R {
00147 #    8.          if q appartient conditions(r)
00148 #    9.           and conditions(r) est inclus dans la base de donnees
00149 #   10.           and conclusion(r) pas dans la base de donnees {
00150 #   11.                  ajouter conclusion(r) en queue de Q
00151 #   12.          }
00152 #   13.  }
00153 #   14. }
00154 # @endcode
00155 #
00156 # @param[in] faitsInitiaux la liste des faits initiaux
00157 # @param[in] regles la liste des regles
00158 def chainageAvantSimple( faitsInitiaux, regles ):
00159 
00160         # copie les faits initiaux dans Q
00161         Q = copy.deepcopy( faitsInitiaux )
00162 
00163         # tant qu'il reste des faits a analyser,
00164         # il faut le faire...
00165         while len(Q)>0:
00166 
00167                 # considere le premier fait de la liste Q
00168                 q = Q.pop(0)
00169 
00170                 # si le fait n'est pas encore dans la base des
00171                 # faits, il faut l'ajouter. cela ne sera vrai
00172                 # que lorsqu'une nouvelle consequence sera
00173                 # inseree dans la liste Q
00174                 if q not in faits:
00175                         ajouteFait( q )
00176                         print q
00177 
00178         # pour que la consequence d'une regle soit ajoutee
00179         # dans la base des faits, il faut respecter les trois
00180         # conditions suivantes:
00181         #
00182                 #        1 - le fait q est une condition de la regle r
00183                 #        2 - les autres conditions de r sont satisfaites par
00184                 #                la base des faits actuelle
00185                 #        3 - la consequence de la regle r n'est pas encore
00186                 #                dans la base des faits
00187                 for r in regles:
00188                         if ( satisfaitUneCondition(r,q) and
00189                                  satisfaitConditions(r) and
00190                                  consequenceRegle(r) not in faits ):
00191                                 Q.append( consequenceRegle(r) )
