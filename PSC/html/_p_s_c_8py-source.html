00048 ## precedentes dans PSC.VARIABLES est satisfaite

00054         for c in CONTRAINTES:

00056

00060

00279 ## Algorithme de backtracking simple
00280 #
00281 # Le principe de l'algorithme est de tenter d'assigner une valeur a chaque variable
00282 # dans la liste PSC.VAR. A chaque assignation, on verifie que toutes le contraintes
00283 # liant la variable courante avec les variables deja instanciees sont satisfaites. 
00284 # Sinon, on procede a un backtrack pour essayer une autre valeur. 
00285 #
00286 # @param[in] k la profondeur de la recherce (commence a 0)
00287 # @param[in] toutesLesSolutions si \b True alors cherche TOUTES les solutions,
00288 #                       sinon s'arrete a la premiere (valeur par defaut)
00289 # @param[in] initialise initialise les variables globales ITERATIONS et SOLUTIONS
00290 # @return instances contenant la solution \e ou echec si aucune solution trouvee
00291 # @see Chapitre 8.4.1 du livre
00292 def backtrack( k, toutesLesSolutions=None, initialise=None ):
00293 
00294         # l'algorithme utilise est le backtracking
00295         algo="bt"
00296 
00297         # variables globales
00298         global ITERATIONS
00299         global SOLUTIONS
00300 
00301         # faut-il ajouter toutes les solutions ?
00302         if toutesLesSolutions==None:
00303                 toutesLesSolutions=False
00304 
00305         # faut-il initialiser les variables ?
00306         if initialise==None or initialise==True:
00307                 ITERATIONS=0
00308                 SOLUTIONS=[]
00309                 libPSC.NBCONTRAINTES=0
00310                 initialise=False
00311 
00312         # incremente le nombre d'iterations et affiche les
00313         # informations du nombre de noeuds visites, etc...
00314         ITERATIONS += 1;
00315         afficheNbIterations( algo, k )
00316 
00317         # si k>nombre-variables, alors on a une solution
00318         # a la profondeur ou l'on se trouve.
00319         if k >= len(VARIABLES):
00320 
00321                 # affiche la solution et l'ajoute dans la liste SOLUTIONS
00322                 solution = {}
00323                 for var in VARIABLES:
00324                         solution[var.nom] = var.valeur
00325                 afficheSolution( algo, solution )
00326                 SOLUTIONS.append( solution )
00327 
00328                 # s'il ne faut pas garder toutes les solutions, on peut
00329                 # quitter l'algorithme a cette etape. La recursion sera
00330                 # deroulee et l'algorithme termine.
00331                 if not toutesLesSolutions:
00332                         return SOLUTIONS
00333 
00334         # sinon, il faut continuer la recherche
00335         else:
00336 
00337                 # obtient la k-ieme variable
00338                 var = VARIABLES[k]
00339 
00340                 # pour chaque valeur du domaine...
00341                 for d in var.domaine:
00342 
00343                         # ... teste la variable avec la valeur d
00344                         var.metAJourValeur(d)
00345 
00346                         # verifie si cette valeur d est consistante avec 
00347                         # toutes les contraintes portant sur la variable k
00348                         # et sur au moins une des variables 0..k-1 
00349                         if consistanceAvecVarsPrecedentes(k):
00350 
00351                                 # continue l'algorithme sur la variable k+1
00352                                 reste = backtrack( k+1, toutesLesSolutions, initialise )
00353                                 if reste!= ECHEC:
00354                                         return reste
00355 
00356         # si l'on arrive ici, cela signifie que les valeurs pour les variables 0 a k-1 ne sont consistantes
00357         # avec aucune valeur possible pour la variable k. 
00358         var.metAJourValeur(None)
00359         return ECHEC
